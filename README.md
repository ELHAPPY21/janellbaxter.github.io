private void Play()
        {
            foreach (TriviaItem triviaItem in Questions)
            {
               Clear();
               WriteLine(player.PlayerInformation());
               WriteLine(triviaItem.Question);
                string input = ReadLine();
                if (triviaItem.CheckAnswer(input))
                {
                    player.IncrementScore();
                }
            }
        }
